    <script>

    let grid = [];

    let wordPositions = [];

    let words = [];


    function generateCrossword() {

        words = document.getElementById('wordInput').value.split('\n').map(word => word.trim().toUpperCase());

        const grade = document.getElementById('gradeSelect').value;

        if (words.length === 0) {

            alert('Please enter at least one word.');

            return;

        }


        const gridSize = Math.max(15, Math.ceil(Math.sqrt(words.length * 15)));

        grid = createEmptyGrid(gridSize);

        wordPositions = createEmptyGrid(gridSize);


        placeWords(grid, wordPositions, words, grade);

        fillEmptyCells(grid);

        displayGrid(grid, false);

    }


    function createEmptyGrid(size) {

        return Array(size).fill().map(() => Array(size).fill(''));

    }


    function placeWords(grid, wordPositions, words, grade) {

        const directions = getDirectionsForGrade(grade);

        words.sort((a, b) => b.length - a.length);


        for (const word of words) {

            let placed = false;

            let attempts = 0;

            while (!placed && attempts < 100) {

                const dir = directions[Math.floor(Math.random() * directions.length)];

                const row = Math.floor(Math.random() * grid.length);

                const col = Math.floor(Math.random() * grid.length);


                if (canPlaceWord(grid, word, row, col, dir)) {

                    placeWord(grid, wordPositions, word, row, col, dir);

                    placed = true;

                }

                attempts++;

            }

        }

    }


    function getDirectionsForGrade(grade) {

        const directions = [

            [0, 1],   // right

            [1, 0],   // down

            [1, 1]    // down-right

        ];

        if (grade >= '3') {

            directions.push([-1, 1]); // up-right

        }

        if (grade >= '4') {

            directions.push([0, -1], [-1, 0]); // left, up

        }

        if (grade === '5') {

            directions.push([-1, -1], [1, -1]); // up-left, down-left

        }

        return directions;

    }


    function canPlaceWord(grid, word, row, col, dir) {

        const [dx, dy] = dir;

        for (let i = 0; i < word.length; i++) {

            const x = row + i * dx;

            const y = col + i * dy;

            if (x < 0 || x >= grid.length || y < 0 || y >= grid.length) {

                return false;

            }

            if (grid[x][y] !== '' && grid[x][y] !== word[i]) {

                return false;

            }

        }

        return true;

    }


    function placeWord(grid, wordPositions, word, row, col, dir) {

        const [dx, dy] = dir;

        for (let i = 0; i < word.length; i++) {

            const x = row + i * dx;

            const y = col + i * dy;

            grid[x][y] = word[i];

            wordPositions[x][y] = true;

        }

    }


    function fillEmptyCells(grid) {

        const letters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';

        for (let i = 0; i < grid.length; i++) {

            for (let j = 0; j < grid.length; j++) {

                if (grid[i][j] === '') {

                    grid[i][j] = letters[Math.floor(Math.random() * letters.length)];

                }

            }

        }

    }


    function displayGrid(grid, showAnswers) {

        const gridElement = document.getElementById('crosswordGrid');

        gridElement.style.gridTemplateColumns = `repeat(${grid.length}, 30px)`;

        gridElement.innerHTML = '';


        for (let i = 0; i < grid.length; i++) {

            for (let j = 0; j < grid.length; j++) {

                const cell = document.createElement('div');

                cell.className = 'cell';

                cell.textContent = grid[i][j];

                cell.onclick = () => toggleHighlight(cell);

                if (showAnswers && wordPositions[i][j]) {

                    cell.classList.add('highlighted');

                }

                gridElement.appendChild(cell);

            }

        }

    }


    function toggleHighlight(cell) {

        cell.classList.toggle('highlighted');

    }


    function getHint() {

        const unsolvedCells = Array.from(document.querySelectorAll('.cell')).filter(

            (cell, index) => {

                const row = Math.floor(index / grid.length);

                const col = index % grid.length;

                return wordPositions[row][col] && !cell.classList.contains('highlighted');

            }

        );

        if (unsolvedCells.length > 0) {

            const hintCell = unsolvedCells[Math.floor(Math.random() * unsolvedCells.length)];

            hintCell.classList.add('highlighted');

        } else {

            alert('No more hints available!');

        }

    }


    function checkPuzzle() {

        const cells = document.querySelectorAll('.cell');

        cells.forEach((cell, index) => {

            const row = Math.floor(index / grid.length);

            const col = index % grid.length;

            if (wordPositions[row][col]) {

                if (cell.classList.contains('highlighted')) {

                    cell.classList.remove('incorrect', 'missed');

                } else {

                    cell.classList.add('missed');

                }

            } else if (cell.classList.contains('highlighted')) {

                cell.classList.add('incorrect');

            }

        });

    }


    function printCrossword() {

        const printOption = confirm('Do you want to print the puzzle with the answer key?\nClick OK for puzzle with answer key, or Cancel for puzzle only.');

        if (printOption) {

            displayGrid(grid, true);

        } else {

            displayGrid(grid, false);

        }

        window.print();

        displayGrid(grid, false);

    }
